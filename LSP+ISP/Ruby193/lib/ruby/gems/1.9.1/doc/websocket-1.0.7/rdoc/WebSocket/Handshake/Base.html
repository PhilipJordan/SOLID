def inspect
  vars = self.instance_variables.map{|v| "#{v}=#{instance_variable_get(v).inspect}"}.join(", ")
  insp = "#{self.class}:0x%08x" % (self.__id__ * 2)
  "<#{insp} #{vars}>"
end

# File lib/websocket/handshake/base.rb, line 62
def uri
  uri =  secure ? "wss://" : "ws://"
  uri << host
  uri << ":#{port}" if port
  uri << path
  uri << "?#{query}" if query
  uri
end
