def putc(an_object)
  self << case an_object
          when Fixnum
            an_object.chr
          when String
            an_object
          else
            raise TypeError, 'putc: Only Fixnum and String supported'
          end
  an_object
end

def puts(*params)
  params << "\n" if params.empty?
  params.flatten.each do |element|
    val = element.to_s
    self << val
    self << "\n" unless val[-1, 1] == "\n"
  end
end
