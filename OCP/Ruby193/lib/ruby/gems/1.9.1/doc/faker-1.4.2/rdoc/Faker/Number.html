def decimal(l_digits, r_digits = 2)
  l_d = self.number(l_digits)
  r_d = self.number(r_digits)
  "#{l_d}.#{r_d}"
end

def hexadecimal(digits)
  hex = ""
  digits.times { hex += rand(15).to_s(16) }
  hex
end
