# File lib/rspec/matchers/built_in/yield.rb, line 73
def matches?(block)
  probe = YieldProbe.probe(block)

  if @expectation_type
    probe.num_yields.send(@expectation_type, @expected_yields_count)
  else
    probe.yielded_once?(:yield_control)
  end
end
